CONFIGURAÇÕES

🔐 Supabase (Banco de Dados)

Gratuito: Crie uma conta em supabase.com, crie um projeto, vá em Settings → API e copie a URL e a chave anon/public. Depois execute o SQL abaixo no SQL Editor.
Ver SQL para criar tabelas
-- Cole isso no SQL Editor do Supabase e clique em RUN CREATE TABLE IF NOT EXISTS categories ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, name text NOT NULL, slug text NOT NULL UNIQUE, icon text DEFAULT 'fas fa-tag', created_at timestamptz DEFAULT now() ); CREATE TABLE IF NOT EXISTS products ( id text PRIMARY KEY, name text NOT NULL, description text, category text NOT NULL, image_url text, image_path text, is_featured boolean DEFAULT false, created_at timestamptz DEFAULT now() ); -- Se a tabela products já existir: -- ALTER TABLE products ADD COLUMN IF NOT EXISTS is_featured boolean DEFAULT false; -- Categorias padrão INSERT INTO categories (name, slug, icon) VALUES ('Chapéus', 'chapeus', 'fas fa-hat-cowboy'), ('Botas', 'botas', 'fas fa-shoe-prints'), ('Cintos', 'cintos', 'fas fa-gem'), ('Camisas', 'camisas', 'fas fa-tshirt'), ('Camisas Femininas', 'camisas-femininas', 'fas fa-user-tie'), ('Calças', 'calcas', 'fas fa-person'), ('Calças Femininas', 'calcas-femininas', 'fas fa-user'), ('Acessórios', 'acessorios', 'fas fa-ring') ON CONFLICT (slug) DO NOTHING;
